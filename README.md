# Primero
primero repositorio para control de versiones

Algunos Comandos en linux

pwd	Ondica la ruta actual

ls -l 	Lista el contenido del directorio

history	Ver el historial de comandos 

clear 	Limpiar la terminal

cd .. 	Change directory

ls -la 	lista de contenido a detalle (carpetas y archivos ocultos)

Flechas	Navegar historial de comandos 

mkdir 	nombre  make directory crea un nuevo directorio

rmdir 	nombre	borra un directorio

exit 	salir

sudo su	privilegios administrativos

cat 	ver el contenido del archivo

sudo bash 	privilegios administrativos variables de entorno root

rm archivo 	eliminar un archivo 

su - 	privilegios administrativos de entorno root
